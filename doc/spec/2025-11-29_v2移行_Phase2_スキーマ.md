```typescript
  shopId: v.id("shops"),

  status: v.string(), // "pending" | "active" | "resigned"

  magicLinkToken: v.optional(v.string()),
  magicLinkExpiresAt: v.optional(v.number()),

  // 管理情報
  invitedBy: v.optional(v.string()), // authId（管理者のauthId）
  resignedAt: v.optional(v.number()),
  resignationReason: v.optional(v.string()),

  // メタ

  isDeleted: v.boolean(),
})
  .index("by_shop", ["shopId"])
  .index("by_email", ["email"])
  .index("by_shop_and_email", ["shopId", "email"])
  .index("by_magic_link_token", ["magicLinkToken"]);
```

## データマイグレーション

### 既存データの扱い

既存の `shopUserBelongings` データを `staffs` に移行する必要がある。

#### マイグレーションスクリプト（参考）

```typescript
// convex/migrations/migrateToStaffs.ts
import { mutation } from "./_generated/server";

export const migrateToStaffs = mutation({
  args: {},
  handler: async (ctx) => {
    // 1. 既存のshopUserBelongingsを取得
    const belongings = await ctx.db.query("shopUserBelongings").collect();

    for (const belonging of belongings) {
      // 2. userIdからユーザー情報を取得してemailを設定
      // ※ 既存ユーザーにemailがない場合はダミー or 手動設定
      const user = await ctx.db.get(belonging.userId);

      // 3. staffsに挿入
      await ctx.db.insert("staffs", {
        shopId: belonging.shopId,
        email: user?.email || `unknown-${belonging.userId}@example.com`,
        displayName: belonging.displayName,
        status: belonging.status,
        skills: [], // 初期値は空
        maxWeeklyHours: belonging.maxWorkingHoursPerMonth
          ? Math.round(belonging.maxWorkingHoursPerMonth / 4)
          : undefined,
        magicLinkToken: belonging.inviteToken,
        magicLinkExpiresAt: belonging.inviteExpiresAt,
        invitedBy: undefined, // 旧データはuserIdなので変換が必要
        resignedAt: belonging.resignedAt,
        resignationReason: belonging.resignationReason,
        memo: belonging.memo,
        workStyleNote: belonging.workStyleNote,
        hourlyWage: belonging.hourlyWage,
        createdAt: belonging.createdAt,
        isDeleted: belonging.isDeleted,
      });
    }

    return { migrated: belongings.length };
  },
});
```

### 注意事項

- **email取得**: 既存ユーザーにemailがない場合の対応が必要
- **invitedBy変換**: `userId` → `authId` への変換が必要
- **maxWorkingHoursPerMonth → maxWeeklyHours**: 月→週への換算（÷4）
