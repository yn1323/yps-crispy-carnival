### 3. convex/shop/mutations.ts 修正

#### addUser → addStaff

```typescript
// Before
export const addUser = mutation({
  args: {
    shopId: v.id("shops"),
    userId: v.id("users"),
    displayName: v.string(),
    role: v.string(),
  },
  // ...
});

// After
export const addStaff = mutation({
  args: {
    shopId: v.id("shops"),
    email: v.string(),
    displayName: v.string(),
  },
  handler: async (ctx, args) => {
    // 同じ店舗に同じemailがないかチェック
    const existing = await ctx.db
      .query("staffs")
      .withIndex("by_shop_and_email", (q) =>
        q.eq("shopId", args.shopId).eq("email", args.email)
      )
      .first();

    if (existing && !existing.isDeleted) {
      throw new Error("このメールアドレスは既に登録されています");
    }

    const staffId = await ctx.db.insert("staffs", {
      shopId: args.shopId,
      email: args.email,
      displayName: args.displayName,
      status: "pending",
      createdAt: Date.now(),
      isDeleted: false,
    });

    return { success: true, staffId };
  },
});
```

### 5. convex/invite/ → convex/magicLink/ リネーム

#### mutations.ts

```typescript
export const generateMagicLink = mutation({
  args: {
    staffId: v.id("staffs"),
    // シフトサイクルIDがあれば紐付け（将来）
  },
  handler: async (ctx, args) => {
    const token = crypto.randomUUID();
    // 有効期限はシフト確定まで（ここでは仮で2週間）
    const expiresAt = Date.now() + 14 * 24 * 60 * 60 * 1000;

    await ctx.db.patch(args.staffId, {
      magicLinkToken: token,
      magicLinkExpiresAt: expiresAt,
    });

    return { token, expiresAt };
  },
});

export const verifyMagicLink = query({
  args: { token: v.string() },
  handler: async (ctx, args) => {
    const staff = await ctx.db
      .query("staffs")
      .withIndex("by_magic_link_token", (q) => q.eq("magicLinkToken", args.token))
      .first();

    if (!staff) return { valid: false, reason: "not_found" };
    if (staff.isDeleted) return { valid: false, reason: "deleted" };
    if (staff.magicLinkExpiresAt && staff.magicLinkExpiresAt < Date.now()) {
      return { valid: false, reason: "expired" };
    }

    return { valid: true, staff };
  },
});

export const invalidateMagicLink = mutation({
  args: { staffId: v.id("staffs") },
  handler: async (ctx, args) => {
    await ctx.db.patch(args.staffId, {
      magicLinkToken: undefined,
      magicLinkExpiresAt: undefined,
    });
    return { success: true };
  },
});
```
